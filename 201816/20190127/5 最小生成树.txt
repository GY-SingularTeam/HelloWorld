#include<stdio.h>
int a[50][50],b[10],c[10],sum=0,n;
void fun(int u)
{
   int min=0,x;
   if(u==n)return;
   else{
   for(int j=1;j<=n;j++)
   if(b[j]!=0){
   for(int i=0;i<u;i++)
    {
	if(a[c[i]][j]!=0)
     {
	 if(min==0){min=j;x=c[i];}
	 else {
	 if(a[c[i]][j]<a[x][min]){
	 min=j;x=c[i];
	 }
	 }
	 }	
	}
  }
  c[u++]=min;
  b[min]=0;
  sum+=a[x][min];
  fun(u);
   }
}
int main()
{
    int z,m,i,j;
    scanf("%d%d",&n,&m);
    for(int u=0;u<m;u++)
    {
	scanf("%d%d",&i,&j);
	scanf("%d",&a[i][j]);
	a[j][i]=a[i][j];
	}
	for(int u=1;u<=n;u++)
	b[u]=u;
	scanf("%d",&z);
	b[z]=0;
	c[0]=z;
	fun(1);
	printf("%d",sum);
	return 0;
}
