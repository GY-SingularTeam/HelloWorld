package hanjiazhoye;

import java.util.Scanner;

public class a20190118 {
public static void main(String[] args) {
	Scanner sr=new Scanner(System.in );
	int a=sr.nextInt();
	int b=sr.nextInt();
	int c=sr.nextInt();
	int i=0;
	for(i=a+b+c;i<=a*b*c;i++){
		if(i%a==0&&i%b==0&&i%c==0)
			break;
	}
	System.out.println(i);
	sr.close();
	sr.close();
}
}
